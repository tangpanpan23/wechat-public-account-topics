-- 异动批次表（批量异动场景）
CREATE TABLE `h_transfer_batches`
(
`id`                     bigint unsigned NOT NULL AUTO_INCREMENT,
`batch_no`               varchar(50)  NOT NULL COMMENT '批次号',
`batch_name`             varchar(100) NOT NULL COMMENT '批次名称',
`batch_type`             tinyint      NOT NULL DEFAULT '1' COMMENT '批次类型 1:手动创建 2:系统自动',
`applicant_workcode`     varchar(20)  NOT NULL DEFAULT '' COMMENT '申请人工号',
`applicant_name`         varchar(50)  NOT NULL DEFAULT '' COMMENT '申请人姓名',
`total_count`            int          NOT NULL DEFAULT 0 COMMENT '总数量',
`success_count`          int          NOT NULL DEFAULT 0 COMMENT '成功数量',
`failed_count`           int          NOT NULL DEFAULT 0 COMMENT '失败数量',
`status`                 tinyint      NOT NULL DEFAULT '1' COMMENT '批次状态 1:处理中 2:已完成 3:部分失败 4:全部失败',
`effective_date`         date                  DEFAULT NULL COMMENT '批次生效日期',
`description`            text                  DEFAULT NULL COMMENT '批次描述',
`error_summary`          text                  DEFAULT NULL COMMENT '错误汇总',
`updated_at`             timestamp    NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
`created_at`             timestamp    NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
`deleted_at`             datetime(3) DEFAULT NULL,
PRIMARY KEY (`id`),
UNIQUE KEY `idx_batch_no` (`batch_no`),
KEY                      `idx_applicant_workcode` (`applicant_workcode`),
KEY                      `idx_effective_date` (`effective_date`),
KEY                      `idx_status` (`status`)
) ENGINE=InnoDB AUTO_INCREMENT=1 COMMENT '异动批次表';

-- 异动批次明细关联表
CREATE TABLE `h_transfer_batch_items`
(
`id`                     bigint unsigned NOT NULL AUTO_INCREMENT,
`batch_id`               bigint unsigned NOT NULL COMMENT '批次ID',
`transfer_id`            bigint unsigned NOT NULL COMMENT '异动记录ID',
`item_status`            tinyint      NOT NULL DEFAULT '1' COMMENT '明细状态 1:待处理 2:处理成功 3:处理失败',
`error_message`          text                  DEFAULT NULL COMMENT '错误信息',
`sort_order`             int          NOT NULL DEFAULT 0 COMMENT '排序',
`updated_at`             timestamp    NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
`created_at`             timestamp    NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
PRIMARY KEY (`id`),
KEY                      `idx_batch_id` (`batch_id`),
KEY                      `idx_transfer_id` (`transfer_id`),
KEY                      `idx_item_status` (`item_status`),
CONSTRAINT `fk_batch_items_batch` FOREIGN KEY (`batch_id`) REFERENCES `h_transfer_batches` (`id`) ON DELETE CASCADE,
CONSTRAINT `fk_batch_items_transfer` FOREIGN KEY (`transfer_id`) REFERENCES `h_personnel_transfers` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=1 COMMENT '异动批次明细表';

-- 异动日志表
CREATE TABLE `h_transfer_logs`
(
`id`                     bigint unsigned NOT NULL AUTO_INCREMENT,
`transfer_id`            bigint unsigned NOT NULL COMMENT '异动记录ID',
`operation_type`         varchar(50)  NOT NULL COMMENT '操作类型',
`operation_user`         varchar(20)  NOT NULL DEFAULT '' COMMENT '操作人工号',
`operation_name`         varchar(50)  NOT NULL DEFAULT '' COMMENT '操作人姓名',
`operation_time`         timestamp    NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '操作时间',
`operation_content`      text                  DEFAULT NULL COMMENT '操作内容',
`before_data`            json                  DEFAULT NULL COMMENT '操作前数据',
`after_data`             json                  DEFAULT NULL COMMENT '操作后数据',
`ip_address`             varchar(45)  NOT NULL DEFAULT '' COMMENT 'IP地址',
`user_agent`             varchar(500) NOT NULL DEFAULT '' COMMENT '用户代理',
`extra`                  json                  DEFAULT NULL COMMENT '扩展数据',
`created_at`             timestamp    NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
PRIMARY KEY (`id`),
KEY                      `idx_transfer_id` (`transfer_id`),
KEY                      `idx_operation_user` (`operation_user`),
KEY                      `idx_operation_time` (`operation_time`),
KEY                      `idx_operation_type` (`operation_type`),
CONSTRAINT `fk_transfer_logs_transfer` FOREIGN KEY (`transfer_id`) REFERENCES `h_personnel_transfers` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=1 COMMENT '异动操作日志表';
